Esquema SQL

Definición exacta de tablas extraída de schema.sql. Sintaxis PostgreSQL con SERIAL y claves foráneas.

schema.sql — empresa
-- ───────────────────────────────────────────────── -- Base de datos: empresa -- Motor: PostgreSQL | Archivo: schema.sql -- ───────────────────────────────────────────────── CREATE DATABASE empresa; -- ── Tabla 1: proveedores ────────────────────────── CREATE TABLE proveedores ( id_proveedor SERIAL PRIMARY KEY, nombre_proveedor VARCHAR(50), telefono VARCHAR(20), correo VARCHAR(80), direccion VARCHAR(80) ); -- ── Tabla 2: productos ─────────────────────────── CREATE TABLE productos ( id_producto SERIAL PRIMARY KEY, nombre VARCHAR(50), marca VARCHAR(50), categoria VARCHAR(50), precio NUMERIC(10,2), stock INT, id_proveedor INT, FOREIGN KEY (id_proveedor) REFERENCES proveedores(id_proveedor) ); -- ── Tabla 3: ventas ────────────────────────────── CREATE TABLE ventas ( id_venta SERIAL PRIMARY KEY, fecha DATE, nombre_cliente VARCHAR(80), documento_cliente VARCHAR(20) ); -- ── Tabla 4: detalle_venta ─────────────────────── CREATE TABLE detalle_venta ( id_detalle SERIAL PRIMARY KEY, id_venta INT, id_producto INT, cantidad INT, FOREIGN KEY (id_venta) REFERENCES ventas(id_venta), FOREIGN KEY (id_producto) REFERENCES productos(id_producto) ); -- ── Consulta JOIN: ventas con producto y proveedor ─ SELECT v.id_venta, v.nombre_cliente, p.nombre AS producto, pr.nombre_proveedor AS proveedor, dv.cantidad FROM detalle_venta dv JOIN ventas v ON dv.id_venta = v.id_venta JOIN productos p ON dv.id_producto = p.id_producto JOIN proveedores pr ON p.id_proveedor = pr.id_proveedor;
